271 static void advance_reg (int *reg_fill)
272 {
273  /* the register steps according to the primitive polynomial */
274  /* (64,4,3,1,0); each call steps register 64 times */
275  /* we use two words to represent the register to allow for integer */
276  /* size of 32 bits */
277 
278  const int mask = 0x1b;
279  int adv_64[4][2];
280  int i, new_fill[2];
281  unsigned temp;
282 
283  adv_64[0][0] = 0xb0000000;
284  adv_64[0][1] = 0x1b;
285  adv_64[1][0] = 0x60000000;
286  adv_64[1][1] = 0x2d;
287  adv_64[2][0] = 0xc0000000;
288  adv_64[2][1] = 0x5a;
289  adv_64[3][0] = 0x80000000;
290  adv_64[3][1] = 0xaf;
291  new_fill[1] = new_fill[0] = 0;
292  temp = mask << 27;
293 
294  for (i = 27; i >= 0; i--)
295  {
296  new_fill[0] = (new_fill[0] << 1) | (1 & bitcnt (reg_fill[0] & temp));
297  new_fill[1] = (new_fill[1] << 1) | (1 & bitcnt (reg_fill[1] & temp));
298  temp >>= 1;
299  }
300 
301  for (i = 28; i < 32; i++)
302  {
303  temp = bitcnt (reg_fill[0] & (mask << i));
304  temp ^= bitcnt (reg_fill[1] & (mask >> (32 - i)));
305  new_fill[0] |= (1 & temp) << i;
306  temp = bitcnt (reg_fill[0] & adv_64[i - 28][0]);
307  temp ^= bitcnt (reg_fill[1] & adv_64[i - 28][1]);
308  new_fill[1] |= (1 & temp) << i;
309  }
310 
311  reg_fill[0] = new_fill[0];
312  reg_fill[1] = new_fill[1];
313 }

315 static void get_fill (uint64 *n, uint64 *r, int param, unsigned seed)
316 {
317  int i, j, k, temp[2], length;
318  uint64 tempui;
319 
320  length = valid[param].L;
321 
322  /* Initialize the shift register with the node number XORed with seed */
323  temp[1] = highword(n[0]);
324  temp[0] = lowword(n[0]) ^ seed;
325  if (!temp[0])
326  temp[0] = GS0;
327 
328  advance_reg (temp); /* Advance the shift register some */
329  advance_reg (temp);
330 
331  /* The first word of the RNG is defined by the LSBs of the node number */
332  And(n[0], INT_MASK, tempui);
333  lshift (tempui, 1, r[0]);
334 
335  /* The RNG is filled with the bits of the shift register, at each time */
336  /* shifted up to make room for the bits defining the canonical form; */
337  /* the node number is XORed into the fill to make the generators unique */
338 
339  for (i = 1; i < length - 1; i++)
340  {
341  advance_reg (temp);
342 
343  seti2(temp[0], temp[1], tempui);
344  Xor(tempui, n[i], tempui);
345  And(tempui, INT_MASK, tempui);
346  lshift (tempui, 1, r[i]);
347  }
348  seti(0, r[length - 1]);
349  /* the canonical form for the LSB is instituted here */
350  k = valid[param].first + valid[param].LSBS;
351 
352  for (j = valid[param].first; j < k; j++)
353  Or(r[j], ONE, r[j]);
354 
355  return;
