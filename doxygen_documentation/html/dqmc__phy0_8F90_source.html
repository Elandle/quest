588  ! Deal with error and expectation values of cv and chi_thermal properly
589  p0%meas(p0_chit, 1:avg) = p0%meas(p0_chit, 1:avg) - p0%n * p0%beta**2 * p0%meas(p0_energy, 1:avg) * p0%meas(p0_density, 1:avg)
590  p0%meas(p0_cv , 1:avg) = p0%meas(p0_cv , 1:avg) - p0%n * (p0%beta * p0%meas(p0_energy, 1:avg))**2
591 
592  p0%meas(p0_cv , err) = sqrt((n-1) * sum((p0%meas(p0_cv , 1:n) - p0%meas(p0_cv , avg))**2))
593  p0%meas(p0_chit, err) = sqrt((n-1) * sum((p0%meas(p0_chit, 1:n) - p0%meas(p0_chit, avg))**2))
594 
595  else
596 
597  mpi_err = 0
598 
599 # ifdef _QMC_MPI
600 
601  n = size(p0%AllProp,1)
602 
603  ! Set bin to 1 and fill the average spin
604  p0%AveSpin(:, 1) = ( p0%SpinZZ(:, 1) + 2.d0*p0%SpinXX(:, 1) ) / 3.d0
605 
606  ! Average sign
607  call mpi_allreduce(p0%sign(:, 1) , p0%sign (:, avg), 3, mpi_double, mpi_sum, mpi_comm_world, mpi_err)
608 
609  ! Average properties
610  call mpi_allreduce(p0%AllProp(:, 1), p0%AllProp(:, avg), n, mpi_double, mpi_sum, mpi_comm_world, mpi_err)
611 
612  ! Compute average over n-1 processors
613  p0%AllProp(:, 1) = (p0%AllProp(:, avg) - p0%AllProp(:, 1)) / dble(nproc - 1)
614  p0%sign (:, 1) = (p0%sign (:, avg) - p0%sign (:, 1)) / dble(nproc - 1)
615 
616  ! Store average amongst all processors
617  p0%AllProp(:, avg) = p0%AllProp(:, avg) / p0%sign(p0_sgn, avg)
618  p0%sign (:, avg) = p0%sign (:, avg) / dble(nproc)
619 
620  ! Store jackknife in the processor bin
621  p0%AllProp(:, 1) = p0%AllProp(:, 1) / p0%sign(p0_sgn, 1)
622 
623  ! Compute proper chi_thermal and C_v
624  p0%meas(p0_chit, 1:avg) = p0%meas(p0_chit, 1:avg) - p0%n * p0%beta**2 * p0%meas(p0_energy, 1:avg) * p0%meas(p0_density, 1:avg)
625  p0%meas(p0_cv , 1:avg) = p0%meas(p0_cv , 1:avg) - p0%n * (p0%beta * p0%meas(p0_energy, 1:avg))**2
626 
627  ! Compute error
628  call mpi_allreduce(p0%AllProp(:, 1)**2, p0%AllProp(:, err), n, mpi_double, mpi_sum, mpi_comm_world, mpi_err)
629  p0%AllProp(:, err) = p0%AllProp(:, err) / dble(nproc) - p0%AllProp(:, avg)**2
630  p0%AllProp(:, err) = sqrt(p0%AllProp(:, err) * dble(nproc-1))
631 
632  ! Compute error for sign
633  call mpi_allreduce(p0%sign(:,1)**2, p0%sign(:,err), 3, mpi_double, mpi_sum, mpi_comm_world, mpi_err)
634  p0%sign(:, err) = p0%sign(:, err) / dble(nproc) - p0%sign(:, avg)**2
635  p0%sign(:, err) = sqrt(p0%sign(:, err) * dble(nproc-1))
636 
637  p0%meas(p0_chit,avg) = p0%meas(p0_chit,avg) - p0%n * p0%beta**2 * p0%meas(p0_energy,avg) * p0%meas(p0_density,avg)
638 
639  p0%meas(p0_cv,avg) = p0%meas(p0_cv,avg) - p0%n * (p0%beta * p0%meas(p0_energy,avg))**2
640 
641 # endif
642 
643  endif
644 
645  end subroutine dqmc_phy0_geterr

648 
649  subroutine dqmc_phy0_meas(n, P0, G_up, G_dn, U, mu_up, mu_dn, t_up, t_dn, sgnup, sgndn, S)
650  !
651  ! Purpose
652  ! =======
653  ! This subroutine performs some physics measurement on
654  ! Hubbard model.
655  !
656  ! Arguments
657  ! =========
658  !
659  integer, intent(in) :: n ! Number of sites
660  type(phy0), intent(inout) :: P0 ! Phy0
661  real(wp), intent(in) :: G_up(n,n) ! Green's function
662  real(wp), intent(in) :: G_dn(n,n) ! for spin up and down
663  real(wp), intent(in) :: sgnup, sgndn ! Sgn for det(G_up) det(G_dn)
664  real(wp), intent(in) :: mu_up(n), mu_dn(n) ! Chemical and Kinetic para
665  real(wp), intent(in) :: t_up(:), t_dn(:) ! Chemical and Kinetic para
666  real(wp), intent(in) :: U(:) ! Chemical and Kinetic para
667  type(struct), intent(in) :: S ! Underline structure
668  target :: s
669 
670  ! ... local scalar ...
671 
672  integer :: i, j, k, ph ! Loop iterator
673  integer :: tmp, idx, m ! Helper variables
674  real(wp) :: sgn
675  real(wp) :: var1, var2, var3
676  integer, pointer :: start(:)
677  integer, pointer :: r(:)
678  integer, pointer :: A(:)
679 
680  ! Auxiliary variable for chi_thermal and C_v
681  real(wp) :: Cbar, Nbar, Tbar, un
682  real(wp) :: h_up(n, n), h_dn(n, n)
683  real(wp) :: A_up(n, n), A_dn(n, n)
684 
685  ! ... executable ...
686 
687  idx = p0%idx
688  tmp = p0%avg
689 
690  ! initialization
691  ! Here we use avg bin as a temp variable
692  p0%meas(:,tmp) = zero
693 
694  p0%G_fun(:,tmp) = zero
695  p0%Gf_up(:,tmp) = zero
696  p0%Gf_dn(:,tmp) = zero
697  p0%Den0(:,tmp) = zero
698  p0%Den1(:,tmp) = zero
699  p0%SpinXX(:,tmp) = zero
700  p0%SpinZZ(:,tmp) = zero
701  p0%Pair(:,tmp) = zero
702 
703  ! Compute the site density for spin up and spin down
704  do i = 1, n
705  !======================================================!
706  ! The density of electrons of spin up(dn) on site i !
707  ! is 1-G_up(i,i) (1-G_dn(i,i)). !
708  ! nup (ndn) is the sum of all spin up (down) electrons.!
709  !======================================================!
710  p0%up(i) = one - g_up(i, i)
711  p0%dn(i) = one - g_dn(i, i)
712  p0%meas(p0_nud, tmp) = p0%meas(p0_nud, tmp) + p0%up(i) * p0%dn(i) * u(s%Map(i))
713 
714  !======================================================!
715  ! Double occupancy P0%up(i) * P0%dn(i)
716  !======================================================!
717  p0%meas(p0_double_occupancy, tmp) = p0%meas(p0_double_occupancy, tmp) + p0%up(i) * p0%dn(i)
718 
719  !=====================================================================!
720  ! Potential energy (P0%up(i)-0.5d0) * (P0%dn(i)-0.5d0) * U(S%Map(i))
721  !=====================================================================!
722  p0%meas(p0_potential_energy, tmp) = p0%meas(p0_potential_energy, tmp) + (p0%up(i) - 0.5d0) * (p0%dn(i) - 0.5d0) * u(s%Map(i))
723  end do
724 
725  p0%meas(p0_nup, tmp) = sum(p0%up)
726  p0%meas(p0_ndn, tmp) = sum(p0%dn)
727 
728  !=================================================================!
729  ! Kinetic energy = tt*sum_{ij\sigma}(G_{ij\sigma}+G_{ji\sigma}) -
730  ! - \sum_{i\sigma} \mu_{i\sigma} (n_{i\sigma} + U_i / 2)!
731  ! where site i and site j are neighbors !
732  !=================================================================!
733  ! Hopping energy = tt*sum_{ij\sigma}(G_{ij\sigma}+G_{ji\sigma})
734 
735  ! set alias
736  start => s%T%cstart
737  r => s%T%row
738  a => s%T%A
739 
740  ! loop all adj sites
741  do i = 1, n ! for each column
742  do j = start(i), start(i + 1)-1 ! for each nonzero elements
743  p0%meas(p0_ke, tmp) = p0%meas(p0_ke, tmp) + t_up(a(j)) * g_up(r(j), i) + t_dn(a(j)) * g_dn(r(j), i)
744  p0%meas(p0_hopping_energy, tmp) = p0%meas(p0_hopping_energy, tmp) + t_up(a(j)) * g_up(r(j), i) + t_dn(a(j)) * g_dn(r(j), i)
745  end do
746  p0%meas(p0_ke, tmp) = p0%meas(p0_ke, tmp) - (mu_up(s%Map(i)) + 0.5d0 * u(s%map(i))) * p0%up(i) - (mu_dn(s%Map(i)) + 0.5d0 * u(s%map(i))) * p0%dn(i)
747  end do
748 
749  !=================================================================!
750  ! Total occupancy = nup + ndn
751  !=================================================================!
752  p0%meas(p0_density, tmp) = p0%meas(p0_nup, tmp) + p0%meas(p0_ndn, tmp)
753 
754  !=================================================================!
755  ! Magnetisation squared = 1/4 (rho - 2 double_occupancy)
756  !=================================================================!
757  p0%meas(p0_magnetization_squared, tmp) = 0.25d0 * (p0%meas(p0_density, tmp) - 2 * p0%meas(p0_double_occupancy, tmp))
758 
759  !=================================================================!
760  ! Total energy = hopping energy + potential energy -
761  ! - sum_{i \sigma} (\mu_{up i \sigma} n_{dn i \sigma})
762  !=================================================================!
763  p0%meas(p0_energy, tmp) = p0%meas(p0_hopping_energy, tmp) + p0%meas(p0_potential_energy, tmp)
764  do i = 1, n
765  p0%meas(p0_energy, tmp) = p0%meas(p0_energy, tmp) - (mu_up(s%Map(i)) * p0%up(i) + mu_dn(s%Map(i)) * p0%dn(i))
766  enddo
767 
768  !=========================================!
769  ! Chi_thermal
770  !=========================================!
771 
772  ! Fill h_up, h_dn with hopping matrix elements
773  h_up = zero
774  h_dn = zero
775  do i = 1, n
776  do j = start(i), start(i + 1) - 1
777  h_up(r(j), i) = -t_up(a(j))
778  h_dn(r(j), i) = -t_dn(a(j))
779  end do
780  h_up(i,i) = h_up(i,i) - mu_up(s%Map(i)) - 0.5d0 * u(s%Map(i))
781  h_dn(i,i) = h_dn(i,i) - mu_dn(s%Map(i)) - 0.5d0 * u(s%Map(i))
782  end do
783 
784  ! Gfun * t
785  ! Fill h_up, h_dn
786  call blas_dgemm('N', 'N', n, n, n, one, g_up, n, h_up, n, zero, a_up, n)
787  call blas_dgemm('N', 'N', n, n, n, one, g_dn, n, h_dn, n, zero, a_dn, n)
788 
789  ! Total number of particles
790  nbar = sum(p0%up) + sum(p0%dn)
791 
792  tbar = 0.d0
793  do i = 1, n
794  tbar = tbar + h_up(i, i) + h_dn(i, i)
795  enddo
796 
797  cbar = 0.d0
798  do i = 1, n
799  cbar = cbar + a_up(i, i) + a_dn(i, i)
800  enddo
801 
802  ! < N T >
803  p0%meas(p0_chit, tmp) = (tbar - cbar) * nbar
804  do j = 1, n
805  do k = 1, n
806  p0%meas(p0_chit, tmp) = p0%meas(p0_chit, tmp) - g_up(j,k) * a_up(k,j) - g_dn(j,k) * a_dn(k,j)
807  enddo
808  enddo
809  p0%meas(p0_chit,tmp) = p0%meas(p0_chit, tmp) + cbar
810 
811  ! < N U >
812  p0%meas(p0_chit,tmp) = p0%meas(p0_chit, tmp) + nbar * p0%meas(p0_nud, tmp)
813  do i = 1, n
814  un = one
815  do k = 1, n
816  un = un - g_up(i, k) * g_up(k, i)
817  enddo
818  p0%meas(p0_chit, tmp) = p0%meas(p0_chit, tmp) + un*p0%dn(i)*u(s%Map(i))
819  un = one
820  do k = 1, n
821  un = un - g_dn(i, k) * g_dn(k, i)
822  enddo
823  p0%meas(p0_chit, tmp) = p0%meas(p0_chit, tmp) + un*p0%up(i)*u(s%Map(i))
824  enddo
825  p0%meas(p0_chit, tmp) = p0%meas(p0_chit, tmp) - two * p0%meas(p0_nud, tmp)
826 
827  ! Scale by inverse temperature
828  p0%meas(p0_chit, tmp) = p0%beta * p0%beta * p0%meas(p0_chit, tmp)
829 
830  !=========================================!
831  ! Specific heat
832  !=========================================!
833 
834  ! < T T >
835  p0%meas(p0_cv, tmp) = (tbar - cbar)**2
836  do i = 1, n
837  do j = 1, n
838  p0%meas(p0_cv, tmp) = p0%meas(p0_cv, tmp) + (h_up(i, j) - a_up(i, j)) * a_up(j, i) + (h_dn(i, j) - a_dn(i, j)) * a_dn(j, i)
839  enddo
840  enddo
841 
842  ! < T U >
843  p0%meas(p0_cv,tmp) = p0%meas(p0_cv,tmp) + (tbar - cbar) * p0%meas(p0_nud, tmp)
844  do i = 1, n
845  ! un = U(S%map(i)) * P0%up(i)
846  un = zero
847  do j = 1, n
848  ! P0%meas(P0_CV,tmp) = P0%meas(P0_CV,tmp) + un * G_dn(j,i) * (h_dn(i,j) - A_dn(i,j))
849  un = un + g_dn(j, i) * (h_dn(i, j) - a_dn(i, j))
850  enddo
851  p0%meas(p0_cv,tmp) = p0%meas(p0_cv,tmp) + un * u(s%map(i)) * p0%up(i)
852  ! un = U(S%map(i)) * P0%dn(i)
853  un = zero
854  do j = 1, n
855  ! P0%meas(P0_CV,tmp) = P0%meas(P0_CV,tmp) + un * G_up(j,i) * (h_up(i,j) - A_up(i,j))
856  un = un + g_up(j, i) * (h_up(i, j) - a_up(i, j))
857  enddo
858  p0%meas(p0_cv, tmp) = p0%meas(p0_cv, tmp) + un * u(s%map(i)) * p0%dn(i)
859  enddo
860 
861 
862  p0%meas(p0_cv, tmp) = p0%meas(p0_cv, tmp) + (tbar - cbar)*p0%meas(p0_nud, tmp)
863  do i = 1, n
864  p0%meas(p0_cv, tmp) = p0%meas(p0_cv, tmp) + u(s%map(i)) * p0%up(i) * a_dn(i,i)
865  ! un = U(S%map(i)) * P0%up(i)
866  un = zero
867  do j = 1, n
868  ! P0%meas(P0_CV,tmp) = P0%meas(P0_CV,tmp) - un * A_dn(i,j) * G_dn(j,i)
869  un = un + a_dn(i,j) * g_dn(j,i)
870  enddo
871  p0%meas(p0_cv,tmp) = p0%meas(p0_cv,tmp) - un * u(s%map(i)) * p0%up(i)
872 
873  p0%meas(p0_cv,tmp) = p0%meas(p0_cv,tmp) + u(s%map(i)) * p0%dn(i) * a_up(i,i)
874  ! un = U(S%map(i)) * P0%dn(i)
875  un = zero
876  do j = 1, n
877  ! P0%meas(P0_CV,tmp) = P0%meas(P0_CV,tmp) - un * A_up(i,j) * G_up(j,i)
878  un = un + a_up(i,j) * g_up(j,i)
879  enddo
880  p0%meas(p0_cv,tmp) = p0%meas(p0_cv,tmp) - un * u(s%map(i)) * p0%dn(i)
881  enddo
882 
883  ! < U U >
884  ! Redefine A_up and A_dn
885  do j = 1, n
886  do i = 1, n
887  un = sqrt(u(s%Map(i))*u(s%Map(j)))
888  a_up(i,j) = un * (p0%up(i)*p0%up(j) - g_up(i,j)*g_up(j,i))
889  a_dn(i,j) = un * (p0%dn(i)*p0%dn(j) - g_dn(i,j)*g_dn(j,i))
890  enddo
891  a_up(j,j) = p0%up(j) * u(s%Map(j))
892  a_dn(j,j) = p0%dn(j) * u(s%Map(j))
893  enddo
894 
895  ! Compute UU contribution to Cv
896  do j = 1, n
897  do i = 1, n
898  p0%meas(p0_cv,tmp) = p0%meas(p0_cv,tmp) + a_up(i,j)*a_dn(i,j)
899  enddo
900  enddo
901 
902  ! Scale by inverse T
903  p0%meas(p0_cv, tmp) = p0%beta * p0%beta * p0%meas(p0_cv, tmp)
904 
905  !=====================!
906  ! Autocorelation term.!
907  !=====================!
908  if (p0%compSAF) then
909  p0%meas(p0_saf, tmp) = two*n - p0%meas(p0_nup, tmp) - p0%meas(p0_ndn, tmp)
910  p0%meas(p0_saf2,tmp) = p0%meas(p0_saf, tmp)
911  end if
912 
913  do i = 1, n
914  do j = 1, n
915  var1 = g_up(i,j) * g_up(j,i) + g_dn(i,j) * g_dn(j,i)
916  var2 = -two * g_up(i,j) * g_dn(j,i)
917  var3 = g_up(i,i) * g_up(j,j) + g_dn(i,i) * g_dn(j,j) - two * g_up(i,i) * g_dn(j,j) - var1
918 
919  ! k is the index
920  k = s%D(i,j)
921  ph = s%gf_phase(i,j)
922  p0%G_fun(k, tmp) = p0%G_fun(k, tmp) + ph*(g_up(i,j) + g_dn(i,j))
923  p0%Gf_up(k, tmp) = p0%Gf_up(k, tmp) + ph*g_up(i,j)
924  p0%Gf_dn(k, tmp) = p0%Gf_dn(k, tmp) + ph*g_dn(i,j)
925  p0%Den0(k, tmp) = p0%Den0(k, tmp) + p0%up(i)*p0%up(j) + p0%dn(i)*p0%dn(j) - var1
926  p0%Den1(k, tmp) = p0%Den1(k, tmp) + p0%up(i)*p0%dn(j)
927  p0%SpinXX(k, tmp) = p0%SpinXX(k, tmp) + var2
928  p0%SpinZZ(k, tmp) = p0%SpinZZ(k, tmp) + var3
929  p0%Pair(k,tmp) = p0%Pair(k,tmp) + g_up(i,j) * g_dn(i,j)
930 
931  if (p0%compSAF) then
932  var1 = s%P(i)*s%P(j)
933  p0%meas(p0_saf, tmp) = p0%meas(p0_saf, tmp) + var1 * var2
934  p0%meas(p0_saf2,tmp) = p0%meas(p0_saf2,tmp) + var1 * var3
935  end if
936  end do
937  ! special case for (i,i)
938  k = s%D(i,i)
939  var1 = g_up(i,i) + g_dn(i,i)
940  p0%Den0(k, tmp) = p0%Den0(k, tmp) + var1
941  p0%SpinXX(k, tmp) = p0%SpinXX(k, tmp) + var1
942  p0%SpinZZ(k, tmp) = p0%SpinZZ(k, tmp) + var1
943  end do
944 
945  p0%meas(p0_sferro, tmp) = sum(p0%SpinXX(:,tmp))
946  p0%meas(p0_sfer2, tmp) = sum(p0%SpinZZ(:,tmp))
947 
948  ! Average
949  p0%meas(:, tmp) = p0%meas(:, tmp) / n
950  do i = 1, p0%nClass
951  p0%G_fun (i, tmp) = p0%G_fun (i, tmp) / s%F(i) * half
952  p0%Gf_up (i, tmp) = p0%Gf_up (i, tmp) / s%F(i)
953  p0%Gf_dn (i, tmp) = p0%Gf_dn (i, tmp) / s%F(i)
954  p0%SpinXX(i, tmp) = p0%SpinXX(i, tmp) / s%F(i)
955  p0%SpinZZ(i, tmp) = p0%SpinZZ(i, tmp) / s%F(i)
956  p0%Den0 (i, tmp) = p0%Den0 (i, tmp) / s%F(i) * half
957  p0%Den1 (i, tmp) = p0%Den1 (i, tmp) / s%F(i)
958  p0%Pair (i, tmp) = p0%Pair(i, tmp) / s%F(i) * half
959  end do
960 
961 
962  if (p0%compSAF) then
963  p0%meas(p0_safsq, tmp) = p0%meas(p0_saf, tmp) * p0%meas(p0_saf, tmp)
964  p0%meas(p0_saf2sq,tmp) = p0%meas(p0_saf2,tmp) * p0%meas(p0_saf2,tmp)
965  end if
966 
967  ! Accumulate result to P0(:, idx)
968  sgn = sgnup * sgndn
969  p0%meas(:, idx) = p0%meas(:, idx) + p0%meas(:, tmp) * sgn
970 
971  m = p0%nClass
972  call blas_daxpy(m, sgn, p0%G_fun (1:m,tmp), 1, p0%G_fun (1:m,idx), 1)
973  call blas_daxpy(m, sgn, p0%Gf_up (1:m,tmp), 1, p0%Gf_up (1:m,idx), 1)
974  call blas_daxpy(m, sgn, p0%Gf_dn (1:m,tmp), 1, p0%Gf_dn (1:m,idx), 1)
975  call blas_daxpy(m, sgn, p0%SpinXX(1:m,tmp), 1, p0%SpinXX(1:m,idx), 1)
976  call blas_daxpy(m, sgn, p0%SpinZZ(1:m,tmp), 1, p0%SpinZZ(1:m,idx), 1)
977  call blas_daxpy(m, sgn, p0%Den0 (1:m,tmp), 1, p0%Den0 (1:m,idx), 1)
978  call blas_daxpy(m, sgn, p0%Den1 (1:m,tmp), 1, p0%Den1 (1:m,idx), 1)
979  call blas_daxpy(m, sgn, p0%Pair (1:m,tmp), 1, p0%Pair (1:m,idx), 1)
980 
981  p0%sign(p0_sgn, idx) = p0%sign(p0_sgn, idx) + sgn
982  p0%sign(p0_sgnup, idx) = p0%sign(p0_sgnup, idx) + sgnup
983  p0%sign(p0_sgndn, idx) = p0%sign(p0_sgndn, idx) + sgndn
984  p0%cnt = p0%cnt + 1
