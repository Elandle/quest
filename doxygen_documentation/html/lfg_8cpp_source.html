42 /* This version has been modified to use two integer-based additive */
43 /* lagged-Fibonacci generators to produce integer, float and double */
44 /* values. The lagged-Fibonacci generators each have 31 bits of */
45 /* precision (after the bit fixed by the canonical form of the */
46 /* generator is removed), 31-bit values are generated by XORing */
47 /* the values after one has been shifted left one bit. The floating */
48 /* point value is formed by dividing the integer by 1.e+32 (the */
49 /* lsb's will be dropped from the mantissa to make room for the */
50 /* exponent), and two of these integer values in sequence are used */
51 /* to get the necessary precision for the double value. */
52 /* */
53 /* This method has the advantage that the generators pass fairly */
54 /* strict randomness tests, including the Birthday Spacings test */
55 /* that additive lagged-Fibonacci generators are well known to */
56 /* fail. The disadvantage is the additional time needed to do the */
57 /* division explicitly, which was avoided in previous versions. */
58 /* (As the division is by powers of 2, the user might well consider */
59 /* making machine-specific versions of this code to insert the bits */
60 /* into the appropriate places and avoid the problem entirely.) */

187 /**************************/
188 /* function advance_reg: */
189 /**************************/
190 static void advance_reg(int *reg_fill)
191 {
192 /* the register steps according to the primitive polynomial */
193 /* (64,4,3,1,0); each call steps register 64 times */
194 /* we use two words to represent the register to allow for integer */
195 /* size of 32 bits */
196 
197  const int mask = 0x1b;
198  int adv_64[4][2];
199 
200  int i,new_fill[2];
201  unsigned temp;
202 
203  adv_64[0][0] = 0xb0000000;
204  adv_64[0][1] = 0x1b;
205  adv_64[1][0] = 0x60000000;
206  adv_64[1][1] = 0x2d;
207  adv_64[2][0] = 0xc0000000;
208  adv_64[2][1] = 0x5a;
209  adv_64[3][0] = 0x80000000;
210  adv_64[3][1] = 0xaf;
211  new_fill[1] = new_fill[0] = 0;
212  temp = mask<<27;
213 
214  for (i=27;i>=0;i--)
215  {
216  new_fill[0] = (new_fill[0]<<1) | (1&bitcnt(reg_fill[0]&temp));
217  new_fill[1] = (new_fill[1]<<1) | (1&bitcnt(reg_fill[1]&temp));
218  temp >>= 1;
219  }
220 
221  for (i=28;i<32;i++)
222  {
223  temp = bitcnt(reg_fill[0]&(mask<<i));
224  temp ^= bitcnt(reg_fill[1]&(mask>>(32-i)));
225  new_fill[0] |= (1&temp)<<i;
226  temp = bitcnt(reg_fill[0]&adv_64[i-28][0]);
227  temp ^= bitcnt(reg_fill[1]&adv_64[i-28][1]);
228  new_fill[1] |= (1&temp)<<i;
229  }
230 
231  reg_fill[0] = new_fill[0];
232  reg_fill[1] = new_fill[1];
233 }

239 static int get_fill(unsigned *n, unsigned *r, int param_local, unsigned seed_local)
240 {
241  int i,j,k,temp[2], length;
242 
243  length = valid[param_local].L;
244 
245 /* initialize the shift register with the node number XORed with */
246 /* the global seed */
247 /* fill the shift register with two copies of this number */
248 /* except when equal to zero */
249  temp[1] = temp[0] = n[0]^seed_local;
250 
251  if (!temp[0])
252  temp[0] = GS0;
253 
254 /* advance the shift register some */
255  advance_reg(temp);
256  advance_reg(temp);
257 
258 /* the first word in the generator is defined by the 31 LSBs of the */
259 /* node number */
260  r[0] = (INT_MASK&n[0])<<1;
261 /* the generator is filled with the lower 31 bits of the shift */
262 /* register at each time, shifted up to make room for the bits */
263 /* defining the canonical form; the node number is XORed into */
264 /* the fill to make the generators unique */
265  for (i=1;i<length-1;i++)
266  {
267  advance_reg(temp);
268  r[i] = (INT_MASK&(temp[0]^n[i]))<<1;
269  }
270  r[length-1] = 0;
271 /* the canonical form for the LSB is instituted here */
272  k = valid[param_local].first + valid[param_local].LSBS;
273 
274  for (j=valid[param_local].first;j<k;j++)
275  r[j] |= 1;
276 
277  return(0);
278 }

310 {
311  unsigned new_val,*r0_local,*r1_local;
312  int hptr_local,lptr,*hp = &hptr;
313  int lv, kv;
314 
315  lv = lval;
316  kv = kval;
317  r0_local = r0;
318  r1_local = r1;
319  hptr_local = *hp;
320  lptr = hptr_local + kv;
321 
322  if (lptr>=lv) lptr -= lv;
323 
324 /* INT_MOD_MASK causes arithmetic to be modular when integer size is */
325 /* different from generator modulus */
326 
327  r0_local[hptr_local] = INT_MOD_MASK&(r0_local[hptr_local] + r0_local[lptr]);
328  r1_local[hptr_local] = INT_MOD_MASK&(r1_local[hptr_local] + r1_local[lptr]);
329  new_val = (r1_local[hptr_local]&(~1)) ^ (r0_local[hptr_local]>>1);
330 
331  if (--hptr_local < 0)
332  hptr_local = lv - 1; /* skip an element in the sequence */
333 
334  if (--lptr < 0)
335  lptr = lv - 1;
336 
337  r0_local[hptr_local] = INT_MOD_MASK&(r0_local[hptr_local] + r0_local[lptr]);
338  r1_local[hptr_local] = INT_MOD_MASK&(r1_local[hptr_local] + r1_local[lptr]);
339  *hp = (--hptr_local < 0) ? lv-1 : hptr_local;
340 
341  return (new_val>>1);
