328  subroutine count_hop_class(lattice, hamilt)

351  hopclass(:) = 0

358  ! Uses symmetry info found in lattice.

402  subroutine count_local_classes(lattice,hamilt)
403  !
404  ! Construct a list of only local classes i.e. the subset of the distance classes defined on the same site.
405  ! Note that we use the symmetry instead of the value to group sites together.
406  ! This is because local classes refers to both U and mu and two sites may have same value of U but different mu.
407  ! If equivalent by symmetry, the two sites have, however, necessarily same u and mu.
408  !

415  real*8 :: muup(lattice%nclass)
416  real*8 :: mudn(lattice%nclass)
417 
418  nlocclass = 0
419  nsites = size(hamilt%tnneig)
420  allocate(hamilt%mylocclass(0:nsites-1))
421 
422  ! Loop over all sites
423  do isite = 0, nsites-1
424  iclass = lattice%myclass(isite, isite)
425  ! Check whether class was already found
426  do jclass = 1, nlocclass
427  if (iclass .eq. localtmp(jclass)) exit
428  enddo
429  ! if not augment the number of local classes

431  nlocclass = nlocclass+1
432  localtmp(jclass) = iclass

434  utmp(jclass) = hamilt%Uv(isite,isite)
435  muup(jclass) = -dble(hamilt%hopup(isite,isite)) + hamilt%mu_up
436  mudn(jclass) = -dble(hamilt%hopdn(isite,isite)) + hamilt%mu_dn
437  endif
438  ! assign site to a class
439  hamilt%mylocclass(isite) = jclass
440  enddo
441 
442  ! Store the number of different classes
443  hamilt%nlocclass = nlocclass
444 
445  ! Store the value of U and on-site energy (shifted by mu) for each class

447  allocate(hamilt%muupvalue(nlocclass))
448  allocate(hamilt%mudnvalue(nlocclass))
449  hamilt%Uvalue(1:nlocclass) = utmp(1:nlocclass)
450  hamilt%muupvalue(1:nlocclass) = muup(1:nlocclass)
451  hamilt%mudnvalue(1:nlocclass) = mudn(1:nlocclass)

457  subroutine dqmc_hamilt_groupckb(hamilt)
458  type(hamiltonian_t), intent(inout) :: hamilt
459  integer :: nh, ickb, ih, jh, kh, ns, hop(3)
460  logical :: skip
461  logical, pointer :: vs(:)
462 
463  ! Allocate array for visited sites (vs)
464  ns = 1 + maxval(hamilt%tckb)
465  allocate(vs(0:ns-1))
466 
467  ! Count the number of hoopings
468  nh = size(hamilt%tckb,2)
469  ickb = 0
470  do ih = 0, nh-1
471  ! Check whether hopping has been already assigned
472  if (hamilt%tckb(3,ih) .gt. 0) cycle
473  ! "ih" is the first of group ickb
474  ickb = ickb + 1
475  ! mark all sites as not visited ...
476  vs(0:ns-1) = .false.
477  ! ... except for those of "ih"
478  vs(hamilt%tckb(1:2,ih)) = .true.
479  ! save ckb group
480  hamilt%tckb(3,ih) = ickb
481  ! Start finding other hopping
482  do jh = 0, nh-1
483  ! If hopping already include or...
484  skip = hamilt%tckb(3,jh) .gt. 0
485  ! if one of the site was already visited then...
486  skip = skip .or. vs(hamilt%tckb(1, jh))
487  skip = skip .or. vs(hamilt%tckb(2, jh))
488  ! do not include "jh" in ickb
489  if (skip) cycle
490  ! otherwise do it and...
491  hamilt%tckb(3,jh) = ickb
492  ! mark the sites as visited
493  vs(hamilt%tckb(1:2, jh)) = .true.
494  enddo
495  enddo
496 
497  deallocate(vs)
498 
499  ! Sort them so that hops in same group are neighbors
500  do ih = 1, nh-1
501  hop(1:3) = hamilt%tckb(1:3, ih)
502  do jh = 0, ih-1
503  if (hop(3) < hamilt%tckb(3, jh)) exit
504  enddo
505  do kh = ih-1, jh, -1
506  hamilt%tckb(1:3, kh+1) = hamilt%tckb(1:3, kh)
507  enddo
508  hamilt%tckb(1:3, jh) = hop(1:3)
509  enddo
