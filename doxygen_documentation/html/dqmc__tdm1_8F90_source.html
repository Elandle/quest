621  do j = 1, t1%properties(igfup)%n
622  ! k is the distance index of site i and site j
623  k = t1%properties(igfup)%D(i, j)
624  value1(k) = value1(k) + 2*factor*upt0(i, j)
625  value2(k) = value2(k) - 2*factor*up0t(i, j)
626  end do
627  end do

629  value1 => t1%properties(igfdn)%values(:, dt1, t1%tmp)
630  value2 => t1%properties(igfdn)%values(:, dt2, t1%tmp)

632  do j = 1, t1%properties(igfdn)%n
633  ! k is the distance index of site i and site j
634  k = t1%properties(igfdn)%D(i,j)
635  value1(k) = value1(k) + 2 * factor * dnt0(i, j)
636  value2(k) = value2(k) - 2 * factor * dn0t(i, j)
637  end do
638  end do

641  value2 => t1%properties(ispxx)%values(:, dt2, t1%tmp)
642  do i = 1, t1%properties(ispxx)%n
643  do j = 1, t1%properties(ispxx)%n
644  ! k is the distance index of site i and site j
645  k = t1%properties(ispxx)%D(i, j)
646  value1(k) = value1(k) - (up0t(j, i) * dnt0(i, j) + dn0t(j, i) * upt0(i, j)) / 2
647  value2(k) = value2(k) - (up0t(i, j) * dnt0(j, i) + dn0t(i, j) * upt0(j, i)) / 2
648  end do
649  end do
650 
651  value1 => t1%properties(ispzz)%values(:, dt1, t1%tmp)
652  value2 => t1%properties(ispzz)%values(:, dt2, t1%tmp)
653  do i = 1, t1%properties(ispzz)%n
654  do j = 1, t1%properties(ispzz)%n
655  ! k is the distance index of site i and site j
656  k = t1%properties(ispzz)%D(i, j)
657  value1(k) = value1(k) - (up0t(j, i) * upt0(i, j) + dn0t(j, i) * dnt0(i, j) - (uptt(i, i) - dntt(i, i)) * (up00(j, j) - dn00(j, j))) * 0.5_wp
658  value2(k) = value2(k) - (dn0t(i, j) * dnt0(j, i) + up0t(i, j) * upt0(j, i) - (uptt(j, j) - dntt(j, j)) * (up00(i, i) - dn00(i, i))) * 0.5_wp
659  end do
660  end do
661 
662  value1 => t1%properties(idens)%values(:, dt1, t1%tmp)
663  value2 => t1%properties(idens)%values(:, dt2, t1%tmp)
664  do i = 1, t1%properties(idens)%n
665  do j = 1, t1%properties(idens)%n
666  ! k is the distance index of site i and site j
667  k = t1%properties(idens)%D(i, j)
668  value1(k) = value1(k) - (up0t(j, i) * upt0(i, j) + dn0t(j, i) * dnt0(i, j) - (uptt(i, i) + dntt(i, i)) * (up00(j, j) + dn00(j, j)) - 2.0_wp * (1.0_wp - uptt(i, i) - up00(j, j)) - 2.0_wp * (1.0_wp - dntt(i, i) - dn00(j, j))) * 0.5_wp
669  value2(k) = value2(k) - (dn0t(i, j) * dnt0(j, i) + up0t(i, j) * upt0(j, i) - (uptt(j, j) + dntt(j, j)) * (up00(i, i) + dn00(i, i)) - 2.0_wp * (1.0_wp - up00(i, i) - uptt(j, j)) - 2.0_wp * (1.0_wp - dn00(i, i) - dntt(j, j))) * 0.5_wp
670  end do
671  end do
672 
673  value1 => t1%properties(ipair)%values(:, dt1, t1%tmp)
674  value2 => t1%properties(ipair)%values(:, dt2, t1%tmp)
675  do i = 1, t1%properties(ipair)%n
676  do j = 1, t1%properties(ipair)%n
677  ! k is the distance index of site i and site j
678  k = t1%properties(ipair)%D(i, j)
679  value1(k) = value1(k) + (upt0(i, j) * dnt0(i, j) + dn0t(j, i) * up0t(j, i)) / 2
680  value2(k) = value2(k) + (upt0(j, i) * dnt0(j, i) + dn0t(i, j) * up0t(i, j)) / 2
681  end do
682  end do
683 
684  ! Uniform (q=0) current structure factor. This quantity can be used to get approximated dc conductivity.
685  value1 => t1%properties(icond)%values(:, dt1, t1%tmp)
686  value2 => t1%properties(icond)%values(:, dt2, t1%tmp)
687 
688  ! J-J correlation is following Simone's trick:
689 
690  ! n: the number of sites in one primitive cell
691  ! nk: the number of cells in the cluster
692  ! D: the list of sites making up the primitive links
693  ! np: the number of all links, instead of primitive links
694 
695  ! Double Loops over all links:
696 
697  nsite = t1%properties(icond)%n*t1%properties(icond)%nk
698 
699  ! Call nk the number of unit cells, than the first nk entries of plink
700  ! are the translation of the first primitive link. Likewise from nk+1 to 2nk
701  ! you have all the links obtained by translation of the second link and so on
702 
703  do i = 0, t1%properties(icond)%np - 1
704  do j = 0, t1%properties(icond)%np - 1
705 
706  ! find out if this type of link contribute to conductivity
707  dxl1 = t1%properties(icond)%tlink(3, i)
708  dxl2 = t1%properties(icond)%tlink(3, j)
709 
710  if (abs(dxl1)<1.0e-5 .or. abs(dxl2)<1.0e-5) then
711  value1(1) = value1(1)
712  value2(1) = value2(1)
713  else
714 
715  ! x-coordinate (for JxJx component of J-J tensor) difference
716  dxl1l2 = dxl1*dxl2*0.5_wp/nsite
717  tl1up2up = t1%properties(icond)%tlink(1, i)*t1%properties(icond)%tlink(1, j)
718  tl1dn2dn = t1%properties(icond)%tlink(2, i)*t1%properties(icond)%tlink(2, j)
719  tl1up2dn = t1%properties(icond)%tlink(1, i)*t1%properties(icond)%tlink(2, j)
720  tl1dn2up = t1%properties(icond)%tlink(2, i)*t1%properties(icond)%tlink(1, j)
721 
722  do ic = 0, t1%properties(icond)%nk - 1
723  do jc = 0, t1%properties(icond)%nk - 1
724  l1s = t1%properties(icond)%D(1, i*t1%properties(icond)%nk+ic) + 1
725  l1e = t1%properties(icond)%D(2, i*t1%properties(icond)%nk+ic) + 1
726  l2s = t1%properties(icond)%D(1, j*t1%properties(icond)%nk+jc) + 1
727  l2e = t1%properties(icond)%D(2, j*t1%properties(icond)%nk+jc) + 1
728 
729  ! jx-jx for links l and l':
730 
731  value1(1) = value1(1) - dxl1l2 * tl1up2up * ((uptt(l1e,l1s) - uptt(l1s,l1e)) * (up00(l2e,l2s) - up00(l2s,l2e))) - dxl1l2 * tl1up2dn * ((uptt(l1e,l1s) - uptt(l1s,l1e)) * (dn00(l2e,l2s) - dn00(l2s,l2e))) - dxl1l2 * tl1dn2up * ((dntt(l1e,l1s) - dntt(l1s,l1e)) * (up00(l2e,l2s) - up00(l2s, l2e))) - dxl1l2 * tl1dn2dn * ((dntt(l1e,l1s) - dntt(l1s,l1e)) * (dn00(l2e,l2s) - dn00(l2s,l2e)))
732 
733  ! cross terms
734  value1(1) = value1(1) + dxl1l2 * tl1up2up * (upt0(l1e,l2s) * up0t(l2e,l1s) + upt0(l1s,l2e) * up0t(l2s,l1e) - upt0(l1e,l2e) * up0t(l2s,l1s) - upt0(l1s,l2s) * up0t(l2e,l1e))
735  value1(1) = value1(1) + dxl1l2 * tl1dn2dn * (dnt0(l1e,l2s) * dn0t(l2e,l1s) + dnt0(l1s,l2e) * dn0t(l2s,l1e) - dnt0(l1e,l2e) * dn0t(l2s,l1s) - dnt0(l1s,l2s) * dn0t(l2e,l1e))
736 
737  ! value2: see the rules at the beginning of routine
738  value2(1) = value2(1) - dxl1l2 * tl1up2up * ((up00(l1e,l1s) - up00(l1s,l1e)) * (uptt(l2e,l2s) - uptt(l2s,l2e))) - dxl1l2 * tl1up2dn * ((up00(l1e,l1s) - up00(l1s,l1e)) * (dntt(l2e,l2s) - dntt(l2s,l2e))) - dxl1l2 * tl1dn2up * ((dn00(l1e,l1s) - dn00(l1s,l1e)) * (uptt(l2e,l2s) - uptt(l2s,l2e))) - dxl1l2 * tl1dn2dn * ((dn00(l1e,l1s) - dn00(l1s,l1e)) * (dntt(l2e,l2s) - dntt(l2s,l2e)))
739 
740  ! cross terms
741  value2(1) = value2(1) + dxl1l2 * tl1up2up * (up0t(l1e,l2s) * upt0(l2e,l1s) + up0t(l1s,l2e) * upt0(l2s,l1e) - up0t(l1e,l2e) * upt0(l2s,l1s) - up0t(l1s,l2s) * upt0(l2e,l1e))
742  value2(1) = value2(1) + dxl1l2 * tl1dn2dn * (dn0t(l1e,l2s) * dnt0(l2e,l1s) + dn0t(l1s,l2e) * dnt0(l2s,l1e) - dn0t(l1e,l2e) * dnt0(l2s,l1s) - dn0t(l1s,l2s) * dnt0(l2e,l1e))
743  end do
744  end do
745  endif
746  end do
747  end do
748 
749  else
750 
751  value1 => t1%properties(igfun)%values(:, dt1, t1%tmp)
752  do i = 1, t1%properties(igfun)%n
753  do j = 1, t1%properties(igfun)%n
754  ! k is the distance index of site i and site j
755  k = t1%properties(igfun)%D(i, j)
756  value1(k) = value1(k) + factor * (upt0(i, j) + dnt0(i, j))
757  end do
758  end do
759 
760  value1 => t1%properties(igfup)%values(:, dt1, t1%tmp)
761  do i = 1, t1%properties(igfup)%n
762  do j = 1, t1%properties(igfup)%n
763  ! k is the distance index of site i and site j
764  k = t1%properties(igfup)%D(i, j)
765  value1(k) = value1(k) + 2*factor*upt0(i, j)
766  end do
767  end do
768 
769  value1 => t1%properties(igfdn)%values(:, dt1, t1%tmp)
770  do i = 1, t1%properties(igfdn)%n
771  do j = 1, t1%properties(igfdn)%n
772  ! k is the distance index of site i and site j
773  k = t1%properties(igfdn)%D(i,j)
774  value1(k) = value1(k) + 2*factor*dnt0(i, j)
775  end do
776  end do
777 
778  value1 => t1%properties(ispxx)%values(:, dt1, t1%tmp)
779  do i = 1, t1%properties(ispxx)%n
780  do j = 1, t1%properties(ispxx)%n
781  ! k is the distance index of site i and site j
782  k = t1%properties(ispxx)%D(i, j)
783  value1(k) = value1(k) - up0t(j, i) * dnt0(i, j) - dn0t(j, i) * upt0(i, j)
784  end do
785  end do
786 
787  value1 => t1%properties(ispzz)%values(:, dt1, t1%tmp)
788  do i = 1, t1%properties(ispzz)%n
789  do j = 1, t1%properties(ispzz)%n
790  ! k is the distance index of site i and site j
791  k = t1%properties(ispzz)%D(i, j)
792  value1(k) = value1(k) - (up0t(j, i) * upt0(i, j) + dn0t(j, i) * dnt0(i, j) - (uptt(i, i) - dntt(i, i)) * (up00(j, j) - dn00(j, j)))
793  end do
794  end do
795 
796  value1 => t1%properties(idens)%values(:, dt1, t1%tmp)
797  do i = 1, t1%properties(idens)%n
798  do j = 1, t1%properties(idens)%n
799  ! k is the distance index of site i and site j
800  k = t1%properties(idens)%D(i, j)
801  value1(k) = value1(k) - (up0t(j,i) * upt0(i, j) + dn0t(j, i) * dnt0(i, j) - (uptt(i, i) + dntt(i, i)) * (up00(j, j) + dn00(j, j)) - 2.0_wp * (1.0_wp - uptt(i, i) - up00(j, j)) - 2.0_wp * (1.0_wp - dntt(i, i) - dn00(j, j)))
802  end do
803  end do
804 
805  value1 => t1%properties(ipair)%values(:, dt1, t1%tmp)
806  do i = 1, t1%properties(ipair)%n
807  do j = 1, t1%properties(ipair)%n
808  ! k is the distance index of site i and site j
809  k = t1%properties(ipair)%D(i, j)
810  value1(k) = value1(k) + upt0(i, j) * dnt0(i, j) + dn0t(j, i) * up0t(j, i)
811  end do
812  end do
813 
814  value1 => t1%properties(icond)%values(:, dt1, t1%tmp)
815 
816  nsite = t1%properties(icond)%n*t1%properties(icond)%nk
817 
818  do i = 0, t1%properties(icond)%np - 1
819  do j = 0, t1%properties(icond)%np - 1
820 
821  ! find out if this type of link contribute to conductivity
822  dxl1 = t1%properties(icond)%tlink(3, i)
823  dxl2 = t1%properties(icond)%tlink(3, j)
824  if (abs(dxl1)<1.0e-5 .or. abs(dxl2)<1.0e-5) then
825  value1(1) = value1(1)
826  else
827 
828  dxl1l2 = dxl1*dxl2/nsite
829  tl1up2up = t1%properties(icond)%tlink(1, i) * t1%properties(icond)%tlink(1, j)
830  tl1dn2dn = t1%properties(icond)%tlink(2, i) * t1%properties(icond)%tlink(2, j)
831  tl1up2dn = t1%properties(icond)%tlink(1, i) * t1%properties(icond)%tlink(2, j)
832  tl1dn2up = t1%properties(icond)%tlink(2, i) * t1%properties(icond)%tlink(1, j)
833 
834  do ic = 0, t1%properties(icond)%nk - 1
835  do jc = 0, t1%properties(icond)%nk - 1
836  l1s = t1%properties(icond)%D(1, i*t1%properties(icond)%nk+ic) + 1
837  l1e = t1%properties(icond)%D(2, i*t1%properties(icond)%nk+ic) + 1
838  l2s = t1%properties(icond)%D(1, j*t1%properties(icond)%nk+jc) + 1
839  l2e = t1%properties(icond)%D(2, j*t1%properties(icond)%nk+jc) + 1
840 
841  ! jx-jx for links l and l':
842  ! include x-coordinate (for JxJx component of J-J tensor) difference
843 
844  value1(1) = value1(1) - dxl1l2 * tl1up2up * ((uptt(l1e,l1s) - uptt(l1s,l1e)) * (up00(l2e,l2s) - up00(l2s,l2e))) - dxl1l2 * tl1up2dn * ((uptt(l1e,l1s) - uptt(l1s,l1e)) * (dn00(l2e,l2s) - dn00(l2s,l2e))) - dxl1l2 * tl1dn2up * ((dntt(l1e,l1s) - dntt(l1s,l1e)) * (up00(l2e,l2s) - up00(l2s,l2e))) - dxl1l2 * tl1dn2dn * ((dntt(l1e,l1s) - dntt(l1s,l1e)) * (dn00(l2e,l2s) - dn00(l2s,l2e)))
845 
846  ! cross terms
847  value1(1) = value1(1) + dxl1l2 * tl1up2up * (upt0(l1e,l2s) * up0t(l2e,l1s) + upt0(l1s,l2e) * up0t(l2s,l1e) - upt0(l1e,l2e) * up0t(l2s,l1s) - upt0(l1s,l2s) * up0t(l2e,l1e))
848  value1(1) = value1(1) + dxl1l2 * tl1dn2dn * (dnt0(l1e,l2s) * dn0t(l2e,l1s) + dnt0(l1s,l2e) * dn0t(l2s,l1e) - dnt0(l1e,l2e) * dn0t(l2s,l1s) - dnt0(l1s,l2s) * dn0t(l2e,l1e))
849  end do
850  end do
851  endif
852  end do
853  end do
854  endif

897  !
898  ! Purpose
899  ! =======
900  ! This subroutine compute the error in tdm using the jackknife
901  !
902  ! Arguments
903  ! =========
904  !
905  type(tdm1), intent(inout) :: T1 ! T1
906 
907  ! ... local scalar ...
908  integer :: i, j, iprop
909  integer :: nproc, n, avg, err, mpi_err
910  real(wp) :: sum_sgn, sgn(T1%nBin), y(T1%nBin), data(T1%nBin)
911  real(wp) :: average, error
912 
913  if (.not. t1%compute) return
914  ! ... Executable ...
915  nproc = qmc_sim%size
916  n = t1%nbin
917  avg = t1%avg
918  err = t1%err
919 
920  if (nproc .eq. 1) then
921  data = t1%sgn(1:n)
922  call dqmc_jackknife(n, t1%sgn(avg), t1%sgn(err), data , y, sgn, sum_sgn)
923  do iprop = 1, ntdmarray
924  do i = 1, t1%properties(iprop)%nClass
925  do j = 0, t1%L-1
926  data = t1%properties(iprop)%values(i, j, 1:n)
927  call dqmc_signjackknife(n, average, error, data, y, sgn, sum_sgn)
928  t1%properties(iprop)%values(i, j, avg) = average
929  t1%properties(iprop)%values(i, j, err) = error
930  enddo
931  end do
932  enddo
933  else
934  mpi_err = 0
935 
936  # ifdef _QMC_MPI
937  ! Average sign
938  call mpi_allreduce(t1%sgn(1), t1%sgn(avg), 1, mpi_double, mpi_sum, mpi_comm_world, mpi_err)
939 
940  ! Average properties
941  do iprop = 1, ntdmarray
942  binptr => t1%properties(iprop)%values(:, :, 1)
943  aveptr => t1%properties(iprop)%value(:, :, avg)
944  n = t1%properties(iprop)%nClass * t1%L
945  call mpi_allreduce(binptr, aveptr, n, mpi_double, mpi_sum, mpi_comm_world, mpi_err)
946  enddo
947 
948  ! Compute average over n-1 processors
949  do iprop = 1, ntdmarray
950  binptr => t1%properties(iprop)%values(:, :, 1)
951  aveptr => t1%properties(iprop)%values(:, :, avg)
952  binptr = (aveptr - binptr) / dble(nproc - 1)
953  enddo

955 
956  ! Store average amongst all processors
957  do iprop = 1, ntdmarray
958  aveptr => t1%properties(iprop)%values(:, :, avg)
959  aveptr = aveptr / t1%sgn(avg)
960  enddo

962 
963  ! Store jackknife in the processor bin
964  do iprop = 1, ntdmarray
965  binptr => t1%properties(iprop)%values(:, :, 1)
966  binptr = binptr / t1%sgn(1)
967  enddo

969  ! Compute error
