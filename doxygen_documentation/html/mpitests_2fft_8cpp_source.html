130 
131  /* The "reasonable" permitted distance from the expected value is some */
132  /* multiple of the standard deviation. We choose different multiples */
133  /* for different sample sizes; otherwise the output could be too large */
134  /* or too small. The percentile range corresponds to this multiple; */
135  /* but this value is approximate, especially for small values of 'nruns' */
136  /* since the normality assumption may not hold otherwise. */
137  count = 0;
138  if (nstreams * n < 100) /* percentile at which test fails. */
139  { /* Different sizes have different */
140  mult = 1.0; /* percentiles; otherwise the number */
141  percentile = 68.0; /* of coefficients printed will be too */
142  } /* large. */
143  else if (nstreams * n < 5000)
144  {
145  mult = 2.0;
146  percentile = 95.0;
147  }
148  else if (nstreams * n < 20000)
149  {
150  mult = 3.0;
151  percentile = 99.7;
152  }
153  else
154  {
155  mult = 3.9;
156  percentile = 99.99;
157  }
158 
159  mean1 = (double) nstreams * (double) n / 2.0; /* Expected value of constant term*/
160  mean2 = 0.0; /* Expected values of all other terms */
161  StdDev1 = sqrt ((double) nstreams * (double) n / (double) nruns / 12.0);
162  StdDev2 = sqrt ((double) nstreams * (double) n / (double) nruns / 24.0);
163 
164  if (fabs (means[0] - mean1) > mult * StdDev1) /* Check constant term */
165  {
166  count++;
167  printf (
168  "%d. (0,0) th coefficient exceeds ~ %g th percentile: Expected = %g, observed = %g, StdDev = %g\n\n",
169  count, percentile, mean1, means[0], StdDev1);
170  }
171 
172  for (j = 0; j < nstreams; j++) /* Check each coefficient */
173  for (i = 0; i < lda / 2; i++)
174  {
175  if (i == 0 && j == 0) /* The constant term has already been checked */
176  continue;
177  else
178  {
179  if (nstreams % 2 == 1 || j % (nstreams / 2) != 0
180  || i % (n / 2) != 0)
181  StdDev = StdDev2;
182  else
183  StdDev = StdDev1;
184  }
185 
186  if (fabs (means[2 * i + j * lda] - mean2) > mult * StdDev) /* Real part of coeff. */
187  {
188  count++;
189  printf (
190  "%d. Real part of (%d,%d) th coefficient exceeds ~ %g th percentile: Expected = %g, observed = %g, StdDev = %g\n\n",

192  }
