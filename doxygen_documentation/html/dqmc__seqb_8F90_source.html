158  subroutine dqmc_udtd(n, U, D, T, W1, W2, rw, tau, piv, lwork)
159  !
160  ! Purpose
161  ! =======
162  ! This subroutine computes (updates) UDT decomposition,
163  !
164  ! A = UDT
165  !
166  ! where U is orthonormal, D is diagonal, and T is normalized
167  ! is some way.
168  !
169  ! In input, U is not orthonormal. Therefore,
170  !
171  ! 1. QR-decomposition with pivoting on U
172  !
173  ! [U, R, P] = QRP(U*D)
174  !
175  ! 2. Normalize R by its diagonal elements and set them to D.
176  !
177  ! R = D*R
178  !
179  ! 3. Apply P to T, W = P*T.
180  !
181  ! 4. Multiply R to W to get the new T, T = R*W = R*P*T.
182  !
183  !
184  ! Arguments
185  ! =========
186  !
187  integer, intent(in) :: n
188  real(wp), intent(inout) :: U(n,n)
189  real(wp), intent(inout) :: D(n)
190  real(wp), intent(inout) :: T(n,n)
191  real(wp), intent(inout) :: W1(n,n) ! R-factor in QR factor
192  real(wp), intent(inout) :: W2(n,n) ! working array in pivoting
193  real(wp), intent(inout) :: rw(:) ! working array in QR factor
194  real(wp), intent(inout) :: tau(n) ! working array in QR factor
195  integer, intent(inout) :: piv(:) ! pivoting array in QRD
196  integer, intent(in) :: lwork(:) ! working array in QR
197 
198  ! ... Local variables ...
199  integer :: info, i
200 
201  ! ... Executable ...
202 
203  ! Compute U = U*D
204  call dqmc_scalecol(n, u, d)
205 
206  ! Initial parameters for dgeqp3
207  piv = 0
208  info = 0
209 
210  ! QR factorization with column pivoting
211  call lapack_dgeqp3(n, n, u, n, piv, tau, rw, lwork(la_geqrf), info)
212 
213  if (info .ne. 0) then
214  call dqmc_error("Error: dgeqp3 in dqmc_UDTD.", info)
215  end if
216 
217  ! dgegp3 returns R-factor on the upper triangle of G.
218  ! The lower triangle of G stores "reflectors",
219  ! which is used to reconstruct the Q-factor.
220 
221  ! Move R-factor to R, and normalize it by diagonal elements.
222  ! W1 = U
223  call blas_dcopy(n*n, u, 1, w1, 1)
224 
225  do i = 1, n
226  ! make T upper triangular.
227  w1(i, 1:i-1) = zero
228  ! D = diag(T).
229  d(i) = w1(i, i)
230  if(d(i) .eq. zero) then
231  call dqmc_error("Error: R-factor is singular: dqmc_UDTD.", i)
232  else
233  ! Normalize R's row by its diagonal. R = inv(D)*R
234  call blas_dscal(n-i+1, one/d(i), w1(i, i), n)
235  endif
236  end do
237 
238  ! Compute V = P*V. (W is used as an temporary variable.)
239  do i = 1, n
240  w2(i,1:n) = t(piv(i), 1:n)
241  end do
242 
243  ! Compute V = R*W = R*P*V
244  call blas_dtrmm('L', 'U', 'N', 'U', n, n, one, w1, n, w2, n)
245 
246  ! T = W2
247  call blas_dcopy(n*n, w2, 1, t, 1)
248 
249  ! Generate Q-factor
250  call lapack_dorgqr(n, n, n, u, n, tau, rw, lwork(la_orgqr), info)
251 
252  end subroutine dqmc_udtd
