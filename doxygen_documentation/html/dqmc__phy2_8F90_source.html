702  subroutine dqmc_phy2_wavesymm(S, P2, Symm)
703  ! INPUT
704  ! W(nb,ns) : wave matrix to be analyzed
705  ! nw : number of wave (number of rows of W)
706  ! nb : number of bonds (number of column of W)
707  ! wa(nw) : wave amplitude (expectation value out of QMC)
708  ! maps(nb,ns) : returns how bonds are mapped under symm operations
709  ! ns : number of symmetry operations
710 
711  ! OUTPUT
712  ! wrepr(nw) : irreducible representation of the wave
713  ! wclass(nw) : group of the wave (a group is defined as the set of waves
714  ! belonging to equivalent irreps)
715  ! wlabel : label of the wave (contains character of representation)
716  ! nirrep : number of irreducible representations
717  ! nwclass : number of inequivalent irreducible representations
718 
719  type(struct), intent(inout) :: S
720  type(phy2), intent(inout) :: P2
721  type(symm_operations), intent(in) :: Symm
722 
723  real(wp), pointer :: W(:,:)
724  real(wp), pointer :: wa(:)
725  integer, pointer :: maps(:,:)
726  integer :: iw, ir, jr, is, ib, jw, nirrep, nw, nb, ns
727  integer :: reprlabel(S%nwave),rdim(S%nwave)
728  real(wp) :: newWave(S%n_b), WaveSymm(S%nwave, S%nwave, Symm%nsymm), irrepChar(S%nwave, Symm%nsymm)
729  logical :: irrep(S%nwave)
730 
731  w=>s%W
732  maps=>symm%map_symm_p
733 
734  if (p2%connected) then
735  wa=>p2%M5(:, p2%avg)
736  else
737  wa=>p2%M4(:, p2%avg)
738  endif
739 
740  nw=s%nwave
741  nb=s%n_b
742  ns=symm%nsymm
743 
744  if (.not. associated(s%wrepr)) allocate(s%wrepr(nw))
745 
746  s%wrepr=0
747  nirrep=0
748  rdim=0
749  ! Check degeneracy of eigenvalues to find the dimension of the representation
750  do iw=1,nw
751  if (s%wrepr(iw) == 0) then
752  nirrep=nirrep+1
753  s%wrepr(iw)=nirrep
754  rdim(nirrep)=1
755  do jw=iw+1,nw
756  if (s%wrepr(jw) == 0 .and. abs(wa(iw) - wa(jw)) < 1.d-6) then
757  s%wrepr(jw) = nirrep
758  rdim(nirrep) = rdim(nirrep) + 1
759  endif
760  enddo
761  endif
762  enddo
763 
764  ! Look at how waves transform under the action of symmetry
765  wavesymm = 0.d0
766  do is=1, ns
767  do iw=1, nw
768  ! Apply symmetry "is" to wave "iw"
769  do ib=1,nb
770  newwave(ib) = w(maps(ib, is), iw)
771  enddo
772  ! Dot-product between all waves (jw) and wave "iw" after transformation by "is"
773  ! WaveSymm(iw,jw,is)=0 means "iw", transformed by "is", has no component on "jw"
774  do jw=1, nw
775  wavesymm(iw, jw, is)=sum(w(:, jw) * newwave(:))
776  enddo
777  enddo
778  enddo
779 
780  ! Check whether dimension of representation is compatible with non-zero
781  ! elements in WaveSymm
782  irrep = .true.
783  do ir=1, nirrep
784  do is=1, ns
785  ! Compare all pairs of waves that do not belong to the same irrep.
786  ! If WaveSymm is larger than 0 for any symmetry then what we found was not
787  ! an irrep. Accidental degeneracy.
788  do jw=1,nw
789  if(s%wrepr(jw) == ir) then
790  do iw=1, nw
791  if(s%wrepr(jw) /= ir .and. abs(wavesymm(jw, iw, is)) >1.d-10) irrep(ir) = .false.
792  enddo
793  endif
794  enddo
795  enddo
796  enddo
797 
798  ! Compute character of each irreducible representation
799  irrepchar = 0.d0
800  do ir=1, nirrep
801  if (irrep(ir)) then
802  do is=1, ns
803  do iw=1, nw
804  if(s%wrepr(iw) == ir) irrepchar(ir, is) = irrepchar(ir, is) + wavesymm(iw, iw, is)
805  enddo
806  enddo
807  endif
808  enddo
809 
810  ! Find which representations are equivalent (have the same character table)
811  if (.not. associated(s%wclass)) allocate(s%wclass(nirrep))
812  s%nwclass = 0
813  reprlabel = 0
814  do ir=1, nirrep
815  if (irrep(ir)) then
816  do jr=1, ir-1
817  if (irrep(jr)) then
818  if (sum((irrepchar(ir, 1:ns) - irrepchar(jr, 1:ns))**2) < 1.d-6) exit
819  endif
820  enddo
821  if (jr == ir) then
822  s%nwclass = s%nwclass + 1
823  s%wclass(ir) = s%nwclass
824  else
825  s%wclass(ir) = s%wclass(jr)
826  endif
827  endif
828  enddo
829 
830  ! Assign to each wave a group (which is made by equivalent representations)
831  do iw=1, nw
832  jr = s%wrepr(iw)
833  write(s%wlabel(iw), '(10(i3))') rdim(jr), (nint(irrepchar(jr, is)), is=1, ns)
834  enddo
835 
836  s%nirrep = nirrep
837 
838  end subroutine dqmc_phy2_wavesymm
