735  nsites = lattice%nsites
736  ndim = lattice%ndim

752  do jbond=1, ibond-1

767  ntotpair = bonds%ntotbond
768  pair_origin(1:ntotpair) = bond_origin(1:ntotpair)
769  pair_target(1:ntotpair) = bond_target(1:ntotpair)
770  pair_label(1:ntotpair) = bonds%bond_label(1:ntotpair)
771  xxpair(:, 1:ntotpair) = bonds%xxbond(:, 1:ntotpair)
772  !... But only if there was no PAIR field specified in input
773  if (.not. found_field(pairs_f)) then
774  ! Loop over all pairs having the first atom inside the unit cell
775  do iat=0, natom-1
776  do jat=0, nsites-1
777  ! If bond is already "on" on this pair cycle
778  if (bond_on(iat, jat)) cycle
779  ! Otherwise loop over classes and found other bonds which belongs to same class
780  do bcl=1, nbclass
781 
782  if (tag(bcl) == lattice%myclass(iat, jat)) then
783  ! We found a bond that needs to be included!
784  ! Assign a label to the bond
785  newlabel = newlabel + 1
786  ! Increase the number of total bonds
787  ntotpair = ntotpair + 1
788  ! Save its attribute
789  pair_origin(ntotpair) = iat
790  pair_target(ntotpair) = jat
791  pair_label(ntotpair) = newlabel
792  xxpair(:, ntotpair) = lattice%cartpos(:, jat) - lattice%cartpos(:, iat)
793  ! make xxpair as small as possible
794  do it=1, rdim
795  proj(it) = sum(xxpair(:, ntotpair) * invscc(it, :))
796  enddo
797  proj(1:ndim) = proj(1:ndim) - nint(proj(1:ndim))
798  do it=1, rdim

802  bond_on(iat, jat) = .true.
803 
804  ! Construct the opposite bond
805  if (iat /= jat) then
806  ntotpair = ntotpair + 1
807  pair_label(ntotpair) = -newlabel
808  ! Find the translation that maps iat inside the unit cell
809  it = symmop%translback(jat)
810  ! Translate jat and iat. jat returns the origin of the bond.
811  pair_origin(ntotpair) = symmop%translate(jat, it)
812  pair_target(ntotpair) = symmop%translate(iat, it)

814  ! Switch on the bond flag
815  bond_on(symmop%translate(jat, it), symmop%translate(iat, it)) = .true.
816  endif
817  ! We switch this bons on. Stop looking over classes.
818  exit
819  endif
820 
821  enddo
822  enddo
823  enddo
824  endif

829  ! First deallocate old stuff
830  deallocate(bonds%bond_origin, bonds%xxbond, bonds%bond_label, bonds%bond_target)
831  ! Allocate and store temporary "pair" variable in "Bonds"
832  allocate(bonds%bond_origin(ntotpair), bonds%bond_target(ntotpair), bonds%xxbond(3,ntotpair), bonds%bond_label(ntotpair))
833  bonds%ntotbond = ntotpair
834  bonds%bond_origin(1:ntotpair) = pair_origin(1:ntotpair)
835  bonds%bond_target(1:ntotpair) = mod(pair_target(1:ntotpair), natom)
836  bonds%xxbond(:, 1:ntotpair) = xxpair(:, 1:ntotpair)
837  bonds%bond_label(1:ntotpair) = pair_label(1:ntotpair)
838  endif
839

841  allocate(map_symm(ntotpair, symmop%nsymm))
842  do isymm=1, symmop%nsymm
843  do ibond=1, ntotpair
844  ! Find the two sites where the bond origin and target are mapped into
845  iat = symmop%map_symm(pair_origin(ibond), isymm)
846  jat = symmop%map_symm(pair_target(ibond), isymm)
847  ! Translate them back do that origin is inside unit cell
848  it = symmop%translback(iat)
849  iat = symmop%translate(iat, it)
850  jat = symmop%translate(jat, it)
851  ! Find which bond is defined by (iat,jat)
852  do jbond=1, ntotpair
853  if (pair_origin(jbond) == iat .and. pair_target(jbond) == jat) exit
854  enddo
855  if (jbond > ntotpair) stop 'Symmetry analysis : Cannot find equivalent bond'
856  ! Save the action of the symmetry operation
857  map_symm(ibond,isymm) = jbond
858  enddo
859  enddo
860  deallocate(pair_origin, pair_target, pair_label, xxpair)
861  symmop%map_symm_b => map_symm
862  symmop%bonds_mapped = .true.

865  ! write(*, *)
866  ! write(*, *) 'Bonds (Set completed using symmetry)'
867  ! do ibond=1, Bonds%ntotbond
868  ! write(*, '(3i4,3f12.7)') ibond, Bonds%bond_label(ibond), Bonds%bond_origin(ibond), Bonds%xxbond(1:rdim, ibond)
869  ! enddo
870  ! write(*, *)

872  ! do isymm=1, SymmOp%nsymm

875  ! write(*, *) ibond, '-->', map_symm(ibond, isymm)
876  ! enddo
877  ! enddo

879 
880  bonds%analyzed=.true.

917  !------------------------------------------------------------------------------
918  ! Construct myclass(i,j). Given to sites i and j (not necessarily different)
919  ! returns the class to which they belong. A class contains pairs of orbitals
920  ! that, because of symmetry, are going to have identical pair-correlation
921  ! functions.
922  ! Returns nclass, the number of classes, and class_size(iclass), the
923  ! number of pairs inside class iclass.
924  ! Returns class_label. This is a 4-components array. The first three are the
925  ! cartesian separation of the two orbitals in the pair. The last component is
926  ! the number of the atom inside the primitive cell that belongs to the pair.
927  !------------------------------------------------------------------------------
928  subroutine construct_lattice_classes(SymmOp,lattice)
929  use dqmc_latt
930  integer :: i, it, ip, is, j, isymm, iclass, istart, csize, csizenew, itransl, jclass, idj, id, mclass, ip_transl, is_transl, ip2, is2, jstart, nclass, nsites, natom, nsymm, ntransl
931  integer, allocatable :: patom(:, :), satom(:,:), csizev(:)
932  integer, pointer :: myclass(:, :)
933  type(symm_operations) :: SymmOp
934  type(lattice_t) :: lattice
935 
936  if (.not. symmop%lattice_mapped) stop 'Need to map symmetries over lattice before classes'
937 
938  ! initialize local variables
939  natom = lattice%natom
940  nsites = lattice%nsites
941  nsymm = symmop%nsymm
942  ntransl = symmop%ntransl
943 
944  ! allocate internal arrays
945  nclass = (natom * (natom + 1)) / 2 + natom * (nsites - natom)
946  allocate(patom(2, nclass), satom(2, nclass), csizev(nclass))
947 
948  ! At the beginning each distance is a separate class and only
949  ! pairs with at least one atom in the primitive cell are considered.
950  ! The pair (patom(ix,iclass) , satom(ix,iclas)) is the ix-th
951  ! element of class "iclass"
952  allocate(myclass(0:nsites-1, 0:nsites-1))
953  nclass = 0
954  do ip=0, natom-1
955  ! first loop over sites inside primitive cell
956  do is=ip, natom-1
957  nclass = nclass+1
958  if(is /= ip) then
959  csizev(nclass) = 2
960  patom(1, nclass) = ip
961  satom(1, nclass) = is
962  patom(2, nclass) = is
963  satom(2, nclass) = ip
964  myclass(ip, is) = nclass
965  myclass(is, ip) = nclass
966  else
967  csizev(nclass) = 1
968  patom(1, nclass) = ip
969  satom(1, nclass) = is
970  myclass(ip, is) = nclass
971  endif
972  enddo
973  do is=natom, nsites-1
974  nclass = nclass + 1
975  patom(1, nclass) = ip
976  satom(1, nclass) = is
977  csizev(nclass) = 1
978  myclass(ip, is) = nclass
979  enddo
980  enddo
981 
982  ! Loop over symmetry operations. The "+1" symm op is pair permutation.
983  do isymm=1, nsymm+1
984  ! do isymm=1,nsymm
985  ! Loop over classes (for the first symm op, classes are made
986  ! of all individual atom pairs in which the first atom lies
987  ! in the primitive cell and the second anywhere inside the supercell)
988  do iclass=1, nclass
989  istart = 1
990  do
991  csize = csizev(iclass)
992  csizenew = csize
993  do id=istart, csize
994  ! Map the atoms in the class under the symm operation
995  if (isymm == nsymm+1) then
996  ip = satom(id, iclass)
997  is= patom(id, iclass)
998  else
999  ip = symmop%map_symm(patom(id, iclass), isymm)
1000  is = symmop%map_symm(satom(id, iclass), isymm)
1001  endif
1002  ! Find the transformed pair translated such that the firts site is inside the primitive cell
1003  itransl = symmop%translback(ip)
1004  ip2 = symmop%translate(ip, itransl)
1005  is2 = symmop%translate(is, itransl)
1006  ! Find the class to which the pair belongs to
1007  jclass = myclass(ip2, is2)
1008  ! if classes are different they need to be merged
1009  if(jclass /= iclass) then
1010  ! all pairs of class jclass are transfered in class "iclass"
1011  jstart = csizenew
1012  csizenew = csizenew + csizev(jclass)
1013  call resize_class()
1014  do idj=1, csizev(jclass)
1015  ip = patom(idj, jclass)
1016  is = satom(idj, jclass)
1017  myclass(ip, is) = iclass
1018  patom(jstart + idj, iclass) = ip
1019  satom(jstart + idj, iclass) = is
1020  enddo
1021  ! Size of jclass is nullified
1022  csizev(jclass) = 0
1023  endif
1024  enddo
1025  ! if class size did not change we have found all the classes equivalent to "iclass"
1026  if (csizenew == csize) exit
1027  ! Update loop bounds to find new equivalence due to newly added elements
1028  istart = csizev(iclass) + 1
1029  csizev(iclass) = csizenew
1030  enddo
1031  enddo
1032  enddo
1033 
1034  ! Assign a class to all the remaining pair of atoms using translational symmetry
1035  ! redifine nclass as the number of final classes
1036  mclass = 0
1037  do i=1, nclass
1038  if (csizev(i) > 0) mclass = mclass + 1
1039  do j=1, csizev(i)
1040  ip = patom(j, i)
1041  is = satom(j, i)
1042  myclass(ip, is) = mclass
1043  do it=1, ntransl-1
1044  ip_transl = symmop%translate(ip, it)
1045  is_transl = symmop%translate(is, it)
1046  myclass(ip_transl, is_transl) = mclass
1047  enddo
1048  enddo
1049  enddo
1050  deallocate(patom, satom, csizev)
1051  nclass = mclass
1052 
1053  ! Define a class label using the distance between the sites
1054  allocate(lattice%class_label(nclass, 5))
1055  do iclass=1, nclass
1056  prim: do i=0, natom-1
1057  super: do j=0, nsites-1
1058  if (myclass(i, j) == iclass) then
1059  lattice%class_label(iclass, 1:3) = lattice%cartpos(1:3, j) - lattice%cartpos(1:3, i)
1060  ! 03/26/2013: added the second orbital index in lattice%class_label(iclass,5)
1061  lattice%class_label(iclass, 4) = dble(i)
1062  lattice%class_label(iclass, 5) = mod(j, natom)
1063  exit prim
1064  endif
1065  enddo super
1066  enddo prim
1067  enddo
1068 
1069  ! Compute the size of each class
1070  allocate(lattice%class_size(nclass))
1071  lattice%class_size(:) = 0
1072  do i=0, nsites-1
1073  do j=0, nsites-1
1074  iclass = myclass(i, j)
1075  lattice%class_size(iclass) = lattice%class_size(iclass) + 1
1076  enddo
1077  enddo
1078 
1079  ! Save parameter in lattice
1080  lattice%nclass = nclass
1081  lattice%myclass => myclass
1082 
1083  lattice%analyzed = .true.
