1085 
1086  real(wp), intent(in) :: value(nclass)
1087  integer, intent(in) :: n
1088  integer, intent(in) :: nclass
1089  integer, intent(in) :: class(n,n)
1090  integer, intent(in) :: phase(n,n)
1091  integer, intent(in) :: na, nk
1092  complex(wp), intent(in) :: ft_wgt(n/na, nk)
1093  complex(wp), intent(out) :: valuek(nk*na*(na+1)/2)
1094

1096  complex(wp) :: U(na,na), phcurr
1097 
1098  nt = n / na
1099  naa = na*(na+1)/2
1100 
1101  ! Loop over inequvalent k-points
1102  do ik = 1, nk
1103  u = 0.d0
1104  ! Compute Fourier transform matrix at "ik" for (ja, ia)
1105  do ia = 1, na
1106  do ja = ia, na
1107  ! sum over translations
1108  do it = 1, nt
1109  do jt = 1, nt
1110  ! Find atom which is the translation of "ja" by "it"
1111  i = (it-1) * na + ia
1112  j = (jt-1) * na + ja
1113  ! Use class that corresponds to the (i, j) pair
1114  phcurr = phase(i, j) * value(class(i, j))
1115  u(ja, ia) = u(ja, ia) + phcurr * ft_wgt(it, ik) * dconjg(ft_wgt(jt, ik))
1116  enddo
1117  enddo
1118  enddo
1119  enddo
1120 
1121  ! Pointer to Fourier transform of "ip" at "ik" in "ibin"
1122  i = (ik-1) * naa
1123  do ia = 1, na
1124  do ja = ia, na
1125  i = i + 1
1126  valuek(i) = u(ja, ia)
1127  enddo
1128  enddo
1129 
1130  enddo ! Loop over k-points
